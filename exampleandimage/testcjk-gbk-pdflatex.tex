\def\testCJK#1{%
  \begingroup
  \global\CJKfalse
  \def\CJKsymbol##1{\global\CJKtrue}
  \setbox0=\hbox{#1}%
  \endgroup}
  
\providetoggle{ifCJKforgbt}

  \global\togglefalse{ifCJKforgbt}
  \def\CJKsymbol##1{\global\toggletrue{ifCJKforgbt}}
  \setbox0=\hbox{#1}%
  \endgroup
}%
